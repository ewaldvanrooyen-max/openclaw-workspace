## Weekly Deficit Check Script

```python
#!/usr/bin/env python3
"""
ContentEngen Weekly Deficit Check
Scans GCS/R2 and compares against manifest
"""
import os
import csv
from google.cloud import storage

MANIFEST_FILE = "emerald_manifest_6000.csv"
GCS_BUCKET = "contentengen-vault"
R2_BUCKET = "contentengen-edge"

def check_inventory():
    """Scan storage and compare against manifest"""
    # Load manifest
    manifest = load_manifest(MANIFEST_FILE)
    
    # Scan GCS
    gcs_files = scan_gcs(GCS_BUCKET)
    
    # Scan R2
    r2_files = scan_r2(R2_BUCKET)
    
    # Find missing
    existing = set(gcs_files) | set(r2_files)
    missing = [f for f in manifest if f not in existing]
    
    # Output CSV
    with open('/tmp/missing_assets.csv', 'w', newline='') as f:
        writer = csv.writer(f)
        writer.writerow(['filename', 'type', 'volume', 'status'])
        for asset in missing:
            writer.writerow([asset, 'MISSING', '', 'PENDING'])
    
    print(f"Missing: {len(missing)} assets")
    return missing
```
